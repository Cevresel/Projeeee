// LocalStorage'dan verileri alır ve sayfada görüntüler window.onload = function() { var wasteData = JSON.parse(localStorage.getItem('wasteData')) || []; var dataDisplay = document.getElementById('dataDisplay'); if (wasteData.length === 0) { dataDisplay.innerHTML = '

Henüz veri girişi yapılmadı.

'; } else { wasteData.forEach(function(entry) { var div = document.createElement('div'); div.classList.add('data-entry'); div.innerHTML = `

Okul: ${entry.school}

Atık Türü: ${entry.wasteType}

Atık Miktarı: ${entry.weight} kg

Tarih: ${entry.date}

`; dataDisplay.appendChild(div); }); } };
